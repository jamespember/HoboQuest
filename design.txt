namespace hoboquest {

	class Container {
		map<string, *Item> items;

		bool has(Item a);
		bool pickup(Item a);
		Item & drop(Item a);
	}

	class Environment : Container {
		string description;
		map<string, *Environment> neighbours;

		void on_enter(Actor a);
		void on_exit(Actor a);
		void on_drop(Item a);
		void on_pickup(Item a);
	};

	class Actor : Container {
		string name;
		int hp;
		Environment *position;
	}

}
